## FIX 2 — Rate Limiting on All AI Endpoints

**Why**: Without rate limiting, a single user (or attacker) can exhaust your entire Anthropic API budget.

1. Install:
```bash
pnpm --filter @workspace/api add express-rate-limit
```

2. Create `artifacts/api-server/src/middleware/rateLimiter.ts`:

```typescript
import rateLimit from 'express-rate-limit';

// AI endpoints: 20 requests per user per 15 minutes
export const aiRateLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 20,
  keyGenerator: (req) => {
    // Use Clerk userId if authenticated, else IP
    return (req as any).auth?.userId || req.ip || 'anonymous';
  },
  handler: (req, res) => {
    res.status(429).json({
      error: 'Too many requests',
      message: 'You have reached the AI message limit. Please wait 15 minutes.',
      retryAfter: Math.ceil(15 * 60)
    });
  },
  standardHeaders: true,
  legacyHeaders: false
});

// General API: 100 requests per minute per user
export const generalRateLimiter = rateLimit({
  windowMs: 60 * 1000,
  max: 100,
  keyGenerator: (req) => (req as any).auth?.userId || req.ip || 'anonymous',
  standardHeaders: true,
  legacyHeaders: false
});

// Admin endpoints: 30 requests per minute
export const adminRateLimiter = rateLimit({
  windowMs: 60 * 1000,
  max: 30,
  keyGenerator: (req) => (req as any).auth?.userId || req.ip || 'anonymous',
  standardHeaders: true,
  legacyHeaders: false
});
```

3. Apply in the main server file (`artifacts/api-server/src/index.ts`). Add before route definitions:

export const dpdpConsentsTable = pgTable('dpdp_consents', {

import { Request, Response, NextFunction } from 'express';
import { db } from '../db';
import { dpdpConsentsTable } from '@workspace/db/schema';
import { and, eq } from 'drizzle-orm';

export function requireConsent(purpose: string) {
  return async (req: Request, res: Response, next: NextFunction) => {
    const userId = (req as any).auth?.userId;
    if (!userId) return next(); // Auth middleware will handle unauthenticated

    const consent = await db.select().from(dpdpConsentsTable)
      .where(and(
        eq(dpdpConsentsTable.userId, userId),
        eq(dpdpConsentsTable.purpose, purpose),
        eq(dpdpConsentsTable.granted, true)
      )).limit(1);

    if (consent.length === 0) {
      return res.status(403).json({
        error: 'CONSENT_REQUIRED',
        purpose,
        message: `You must grant consent for "${purpose}" to use this feature.`,
        consentUrl: '/settings/privacy'
      });
    }
    next();
  };
}
```

router.post('/verify', requireAuth(), async (req, res) => {
  const userId = (req as any).auth.userId;
  const { registrationNumber, registrationCouncil, specialty, documentUrls } = req.body;

  if (!registrationNumber || !registrationCouncil || !specialty || !documentUrls?.length) {

    .update(body)
    .digest('hex');

  if (expectedSignature !== razorpay_signature) {
    return res.status(400).json({ error: 'Payment verification failed — invalid signature' });
  }

  // Find the pending payment
  const [payment] = await db.select().from(paymentsTable)
    .where(eq(paymentsTable.razorpayOrderId, razorpay_order_id)).limit(1);

  if (!payment || payment.status === 'paid') {
    return res.status(409).json({ error: 'Payment not found or already processed' });
  }

  // Verify this payment belongs to the authenticated user
  if (payment.userId !== userId) {
    return res.status(403).json({ error: 'Unauthorized' });
  }

  // Mark payment as paid + award credits atomically
  await db.transaction(async (tx) => {
    await tx.update(paymentsTable)
      .set({ status: 'paid', razorpayPaymentId: razorpay_payment_id, paidAt: new Date() })
      .where(eq(paymentsTable.razorpayOrderId, razorpay_order_id));

    await tx.update(usersTable)
      .set({ healthCredits: sql`health_credits + ${payment.creditsAwarded}` })
      .where(eq(usersTable.clerkId, userId));
  });

  res.json({
    success: true,
    creditsAwarded: payment.creditsAwarded,
    message: `${payment.creditsAwarded} health credits added to your account!`
  });
});

// Razorpay webhook (for payment events from Razorpay servers)
router.post('/webhook', async (req, res) => {
  const webhookSecret = process.env.RAZORPAY_WEBHOOK_SECRET!;
  const signature = req.headers['x-razorpay-signature'];
  const body = JSON.stringify(req.body);

  const expectedSignature = crypto
    .createHmac('sha256', webhookSecret)
    .update(body)
    .digest('hex');

  if (signature !== expectedSignature) {
    return res.status(400).json({ error: 'Invalid webhook signature' });
  }

  const event = req.body;
  if (event.event === 'payment.failed') {
    const orderId = event.payload?.payment?.entity?.order_id;
    if (orderId) {
      await db.update(paymentsTable)
        .set({ status: 'failed' })
        .where(eq(paymentsTable.razorpayOrderId, orderId));
    }
  }

  res.json({ status: 'ok' });
});

export default router;
```

## FIX 10 — Admin Security (Basic Hardening)

**Why**: Admin Godmode has no 2FA or IP restriction — one compromised account = full platform access.

1. Create `artifacts/api-server/src/middleware/requireAdmin.ts` if it doesn't already exist:

```typescript
import { Request, Response, NextFunction } from 'express';
import { db } from '../db';
import { usersTable } from '@workspace/db/schema';
import { eq } from 'drizzle-orm';

export async function requireAdmin(req: Request, res: Response, next: NextFunction) {
  const userId = (req as any).auth?.userId;
  if (!userId) return res.status(401).json({ error: 'Unauthenticated' });

  const [user] = await db.select({ role: usersTable.role })
    .from(usersTable)
    .where(eq(usersTable.clerkId, userId))
    .limit(1);

  if (!user || user.role !== 'admin') {
    // Log unauthorized admin access attempt
    await db.insert(auditLogsTable).values({
      action: 'UNAUTHORIZED_ADMIN_ACCESS',
      performedBy: userId,
      metadata: { path: req.path, method: req.method, ip: req.ip }
    });
    return res.status(403).json({ error: 'Admin access required' });
  }

  next();
}
```

## FIX 11 — Hinglish Support (Roman-script Hindi detection)

**Why**: Devanagari script detection misses Hinglish — the dominant mobile typing mode in India ("mera sir dard kar raha hai" written in Roman script).

// Expanded Hindi detection — covers both Devanagari AND common Roman-script Hinglish
const HINGLISH_ROMAN_PATTERNS = [
  /\b(dard|bukhar|sir dard|pet dard|seena|khasi|buzar|beemar|dawai|doctor|aspatal)\b/i,
  /\b(mera|meri|mujhe|mere|hum|kya|kaise|kab|kyun|aur|nahi|hai|ho|kar)\b/i,
  /\b(sar|sar dard|sar dard ho|ulti|chakkar|kamzori|thakaan|neend)\b/i,
  /\b(khaana|pani|dawai|tablet|injection|test|khoon|sugar|bp)\b/i
];

const DEVANAGARI_PATTERN = /[\u0900-\u097F]/;

export function detectLanguage(text: string): 'hi' | 'en' {
  if (DEVANAGARI_PATTERN.test(text)) return 'hi';
  const hinglishMatches = HINGLISH_ROMAN_PATTERNS.filter(p => p.test(text)).length;
  return hinglishMatches >= 2 ? 'hi' : 'en';
}
